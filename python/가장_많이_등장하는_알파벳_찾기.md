```python

from collections import Counter

result=[]
my_str = input().strip()
my_str = list(my_str)
my_strs = Counter(my_str)
my_list = list(map(int,my_strs.values()))
my_list.sort(reverse=True)
a = my_list[0]
for key in my_strs:
    if a == my_strs[key]:
        result.append(key)
    else:
        continue
result.sort()
print("".join(result))
```
